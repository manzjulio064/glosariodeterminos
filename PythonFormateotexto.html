# Definir una cadena de texto
texto = "    ¡hOlA, mUnDo!    "

# Eliminar espacios en blanco al inicio y al final de la cadena
texto = texto.strip()

# Convertir todo el texto a mayúsculas
texto = texto.upper()

# Imprimir la longitud del texto después de eliminar espacios en blanco y convertir a mayúsculas
print("Longitud del texto después de eliminar espacios en blanco y convertir a mayúsculas:", len(texto))

# Convertir la primera letra del texto a mayúscula
texto = texto.capitalize()

# Reemplazar la letra 'O' por 'X'
texto = texto.replace('O', 'X')

# Imprimir el texto modificado
print("Texto modificado:", texto)

# Convertir todo el texto a minúsculas
texto = texto.lower()

# Imprimir el texto en minúsculas
print("Texto en minúsculas:", texto)

# Imprimir la cantidad de caracteres en el texto
print("Cantidad de caracteres en el texto:", len(texto))

# Imprimir el primer carácter del texto
print("Primer carácter del texto:", texto[0])

# Imprimir el último carácter del texto
print("Último carácter del texto:", texto[-1])
