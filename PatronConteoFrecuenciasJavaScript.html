Patrón de Conteo de Frecuencias es una técnica muy útil en programación para contar la ocurrencia de elementos en estructuras de datos como arrays o cadenas. Aquí te presento varias implementaciones en JavaScript Node.js con ejemplos detallados:

1. Contador de Votos

javascript
const votos = ['A', 'B', 'A', 'C', 'B', 'A', 'B', 'C'];

const contarVotos = (votos) => {
  const contador = {};
  votos.forEach(voto => {
    contador[voto] = (contador[voto] || 0) + 1;
  });
  return contador;
};

console.log(contarVotos(votos));
// Output: { A: 3, B: 3, C: 2 }

4. Agrupación de Datos

javascript
const productos = [
  { tipo: 'fruta', nombre: 'manzana' },
  { tipo: 'verdura', nombre: 'zanahoria' },
  { tipo: 'fruta', nombre: 'plátano' },
  { tipo: 'fruta', nombre: 'naranja' },
  { tipo: 'verdura', nombre: 'brocoli' }
];

const agruparPorTipo = (productos) => {
  const agrupados = {};
  productos.forEach(producto => {
    if (!agrupados[producto.tipo]) {
      agrupados[producto.tipo] = [];
    }
    agrupados[producto.tipo].push(producto.nombre);
  });
  return agrupados;
};

console.log(agruparPorTipo(productos));
// Output: { fruta: ['manzana', 'plátano', 'naranja'], verdura: ['zanahoria', 'brocoli'] }

7. Comparar la frecuencia de caracteres en dos cadenas

javascript
const cadena1 = 'aabcc';
const cadena2 = 'adcaa';

const compararFrecuencia = (cadena1, cadena2) => {
  const contar = (cadena) => {
    const contador = {};
    for (let char of cadena) {
      contador[char] = (contador[char] || 0) + 1;
    }
    return contador;
  };

  const frecuencia1 = contar(cadena1);
  const frecuencia2 = contar(cadena2);

  const resultado = {};
  for (let char in frecuencia1) {
    if (frecuencia2[char]) {
      resultado[char] = Math.min(frecuencia1[char], frecuencia2[char]);
    }
  }
  return resultado;
};

console.log(compararFrecuencia(cadena1, cadena2));
// Output: { a: 2, c: 1 }

8. Usando Map para contar la frecuencia de elementos

javascript
const elementos = ['x', 'y', 'x', 'z', 'x', 'y'];

const contarConMap = (elementos) => {
  const contador = new Map();
  elementos.forEach(elemento => {
    contador.set(elemento, (contador.get(elemento) || 0) + 1);
  });
  return contador;
};

console.log(contarConMap(elementos));
// Output: Map(3) { 'x' => 3, 'y' => 2, 'z' => 1 }
