function contarElementosRepetidos(n) {
  const contador = {};

  for (const num of n) {
      contador[num] = (contador[num] || 0) + 1;
  }

  let cantidadRepetidos = 0;
  for (const num in contador) {
      if (contador[num] > 1) {
          cantidadRepetidos++;
      }
  }

  return cantidadRepetidos;
}

console.log(contarElementosRepetidos([1, 3, 3, 4, 4, 4]));
